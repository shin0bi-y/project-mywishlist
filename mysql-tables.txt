DBName : mywishlist

- Table Users :
	name Varchar(255)
	email Varchar(255) Primary Key
	password Varchar(255)

- Table List :
	idList int PRIMARY KEY | AUTO INC
	listName Varchar(255)
	idAuthor int
	description Varchar(1024)
	creationDate Date
	limitDate Date

- Table Item :
	idItem int PRIMARY KEY | AUTO INC
	idList int FOREIGN KEY
	itemName Varchar(255)
	description Varchar(1024)
	photoPath Varchar(255)
	idUser int FOREIGN KEY
	price double

- Table Message :
    idMessage int PRIMARY KEY | AUTO INC
    idList int
    idAuthor int
    message varchar(1023)
    date date

------

Requetes "Create tables" : 

use mywishlist;

show tables;

drop table users;
drop table list;
drop table item;

/*------*/

create table users(
    email VARCHAR(255) Primary Key,
    name VARCHAR(255),
    password VARCHAR(255)
);

insert into users (email, name, password) values ('jean@hotmail.com', 'Jean', '123456');

select * from users;

/*------*/

create table list(
    idList int PRIMARY KEY AUTO_INCREMENT,
    listName VARCHAR(255),
    emailAuthor VARCHAR(255),
    description VARCHAR(1024),
    creationDate DATE,
    limitDate DATE,
    FOREIGN KEY fk_emailAuthor (emailAuthor) references users(email)
);

delete from list;
insert into list (listName, emailAuthor, description, creationDate, limitDate) values ('pour yanis', 'jean@hotmail.com', 'cool', now(), now());
select * from list;

/*------*/

create table item(
    idItem int PRIMARY KEY AUTO_INCREMENT,
    idList int,
    itemName VARCHAR(255),
    description VARCHAR(1024),
    photoPath VARCHAR(255),
    emailUser VARCHAR(255),
    FOREIGN KEY fk_idList (idList) references list(idList),
    FOREIGN KEY fk_emailUser (emailUser) references users(email)
);

insert into item (idList, itemName, description, photoPath, emailUser) values (1, 'ThinkPad', 'Best Laptop', '/images/thinkpad.jpg', 'jean@hotmail.com');

select * from item;

/*------*/

create table message (
    idMessage int PRIMARY KEY AUTO_INCREMENT,
    idList int,
    idAuthor int,
    message varchar(1023),
    date DATE
);
