DBName : mywishlist

- Table Users :
	idUser int PRIMARY KEY | AUTO INC
	name Varchar(255)
	email Varchar(255)
	password Varchar(255)

- Table List :
	idList int PRIMARY KEY | AUTO INC
	listName Varchar(255)
	idAuthor int
	description Varchar(1024)
	creationDate Date
	limitDate Date

- Table Item :
	idItem int PRIMARY KEY | AUTO INC
	idList int FOREIGN KEY
	itemName Varchar(255)
	description Varchar(1024)
	photoPath Varchar(255)
	idUser int FOREIGN KEY
	price double

------

Requetes "Create tables" : 

create table users(
    idUser int PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255),
    email VARCHAR(255),
    password VARCHAR(255)
);

create table list(
    idList int PRIMARY KEY AUTO_INCREMENT,
    listName VARCHAR(255),
    idAuthor int,
    description VARCHAR(1024),
    creationDate DATE,
    limitDate DATE
);

create table item(
    idItem int PRIMARY KEY AUTO_INCREMENT,
    idList int,
    itemName VARCHAR(255),
    description VARCHAR(1024),
    photoPath VARCHAR(255),
    idUser int,
    FOREIGN KEY fk_idList (idList) references list(idList),
    FOREIGN KEY fk_idUser (idUser) references users(idUser)
)
