DBName : mywishlist

- Table Users :
	idUser int PRIMARY KEY | AUTO INC
	name Varchar(255)
	email Varchar(255)
	password Varchar(255)

- Table List :
	idList int PRIMARY KEY | AUTO INC
	listName Varchar(255)
	idAuthor int
	description Varchar(1024)
	creationDate Date
	limitDate Date

- Table Item :
	idItem int PRIMARY KEY | AUTO INC
	idList int FOREIGN KEY
	itemName Varchar(255)
	description Varchar(1024)
	photoPath Varchar(255)
	idUser int FOREIGN KEY
	price double

- Table Message :
    idMessage int PRIMARY KEY | AUTO INC
    idList int
    idAuthor int
    message varchar(1023)
    date date

------

Requetes "Create tables" : 

use mywishlist;

show tables;

drop table users;
drop table list;
drop table item;

/*------*/

create table users(
    idUser int NOT NULL PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255),
    email VARCHAR(255),
    password VARCHAR(255)
);

alter table users MODIFY idUser int NOT NULL AUTO_INCREMENT, AUTO_INCREMENT=1;

insert into users (name, email, password) values ('Jean', 'jean@hotmail.com', '123456');

select * from users;

/*------*/

create table list(
    idList int PRIMARY KEY AUTO_INCREMENT,
    listName VARCHAR(255),
    idAuthor int,
    description VARCHAR(1024),
    creationDate DATE,
    limitDate DATE,
    FOREIGN KEY fk_idAuthor (idAuthor) references users(idUser)
);

alter table list MODIFY idList int NOT NULL AUTO_INCREMENT, AUTO_INCREMENT=1;

insert into list (listName, idAuthor, description, creationDate, limitDate) values ('pour yanis', 1, 'cool', now(), now());
select * from list;

/*------*/

create table item(
    idItem int PRIMARY KEY AUTO_INCREMENT,
    idList int,
    itemName VARCHAR(255),
    description VARCHAR(1024),
    photoPath VARCHAR(255),
    idUser int,
    FOREIGN KEY fk_idList (idList) references list(idList),
    FOREIGN KEY fk_idUser (idUser) references users(idUser)
);

alter table item MODIFY idItem int NOT NULL AUTO_INCREMENT, AUTO_INCREMENT=1;
insert into item (idList, itemName, description, photoPath, idUser) values (1, 'ThinkPad', 'Best Laptop', '/images/thinkpad.jpg', 2);

select * from item;

/*------*/

create table message (
    idMessage int PRIMARY KEY AUTO_INCREMENT,
    idList int,
    idAuthor int,
    message varchar(1023),
    date DATE
);
