DBName : mywishlist

- Table Users :
	idUser int PRIMARY KEY | AUTO INC
	name Varchar(255)
	email Varchar(255)
	password Varchar(255)

- Table List :
	idList int PRIMARY KEY | FOREIGN KEY | AUTO INC
	listName Varchar(255)
	idAuthor int
	description Varchar(1024)
	creationDate Date
	limitDate Date

- Table Item :
	idItem int PRIMARY KEY | AUTO INC
	idList int FOREIGN KEY
	itemName Varchar(255)
	description Varchar(1024)
	photoPath Varchar(255)
	idUser int FOREIGN KEY
	price double
